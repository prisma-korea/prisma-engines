pub static IRREGULAR_SUFFIX_INFLECTIONS: &[(&str, &str)] = &[
    ("man$", "men"),
    ("([lm])ouse$", "${1}ice"),
    ("tooth$", "teeth"),
    ("goose$", "geese"),
    ("foot$", "feet"),
    ("zoon$", "zoa"),
    ("([csx])is$", "${1}es"),
];

pub static MODERN_CLASSICAL_INFLECTIONS: &[(&str, &str)] = &[
    ("trix$", "trices"),
    ("eau$", "eaux"),
    ("ieu$", "ieux"),
    ("(..[iay])nx$", "${1}nges"),
];

pub static ADDITIONAL_SUFFIX_INFLECTIONS: &[(&str, &str)] = &[
    // The suffixes -ch, -sh, and -ss all take -es in the plural (churches, classes, etc)...
    (r"([cs])h$", "${1}hes"),
    ("ss$", "sses"),
    // Certain words ending in -f or -fe take -ves in the plural (lives, wolves, etc)...
    ("([aeo]l)f$", "${1}ves"),
    ("([^d]ea)f$", "${1}ves"),
    ("(ar)f$", "${1}ves"),
    ("([nlw]i)fe$", "${1}ves"),
    // Words ending in -y take -ys
    ("([aeiou])y$", "${1}ys"),
    ("y$", "ies"),
];
